## Quick Start

```typescript
import { DodoPayments } from "dodopayments-checkout";

// Initialize the SDK
DodoPayments.Initialize({
  mode: "test", // 'test' or 'live'
  onEvent: (event) => {
    console.log("Checkout event:", event);
  },
  theme: "light", // 'light' or 'dark'
  linkType: "static", // 'static' or 'dynamic'
  displayType: "overlay"
});

// Open checkout
DodoPayments.Checkout.open({
  products: [
    {
      productId: "pdt_your_product_id",
      quantity: 1,
    },
  ],
  redirectUrl: "https://your-website.com/success",
});
```

## Event Handling

The SDK provides real-time events for complete checkout lifecycle monitoring:

<Tabs>
  <Tab title="Basic Event Handler">
    ```typescript
    DodoPayments.Initialize({
      onEvent: (event: CheckoutEvent) => {
        console.log("Event received:", event);
        
        switch (event.event_type) {
          case "checkout.opened":
            console.log("Checkout overlay opened");
            break;
          case "checkout.closed":
            console.log("Checkout overlay closed");
            break;
          case "checkout.redirect":
            console.log("Redirecting to:", event.data?.url);
            break;
          case "checkout.error":
            console.error("Checkout error:", event.data?.message);
            break;
        }
      }
    });
    ```
  </Tab>
  <Tab title="Advanced Event Handler">
    ```typescript
    DodoPayments.Initialize({
      onEvent: (event: CheckoutEvent) => {
        // Log all events for debugging
        console.log("Checkout Event:", event);
        
        switch (event.event_type) {
          case "checkout.opened":
            // Track checkout start in analytics
            gtag('event', 'checkout_started', {
              event_category: 'ecommerce',
              event_label: 'overlay_checkout'
            });
            
            // Show loading indicators
            setCheckoutOpen(true);
            break;
            
          case "checkout.closed":
            // Track checkout abandonment
            gtag('event', 'checkout_abandoned', {
              event_category: 'ecommerce'
            });
            
            // Hide loading indicators
            setCheckoutOpen(false);
            break;
            
          case "checkout.redirect":
            // Track successful payment initiation
            gtag('event', 'payment_initiated', {
              event_category: 'ecommerce',
              redirect_url: event.data?.url
            });
            
            // Optional: Show success message before redirect
            showNotification("Payment processing...", "success");
            break;
            
          case "checkout.error":
            // Log error for debugging
            console.error("Checkout Error Details:", event.data);
            
            // Track errors in analytics
            gtag('event', 'checkout_error', {
              event_category: 'ecommerce',
              error_message: event.data?.message
            });
            
            // Show user-friendly error message
            showNotification(
              event.data?.message || "Payment failed. Please try again.",
              "error"
            );
            
            // Reset UI state
            setCheckoutOpen(false);
            setLoading(false);
            break;
            
          default:
            console.log("Unknown event type:", event.event_type);
        }
      }
    });
    ```
  </Tab>
  <Tab title="React Hook Example">
    ```typescript
    import { useEffect, useCallback } from 'react';
    
    const useCheckoutEvents = () => {
      const [checkoutState, setCheckoutState] = useState({
        isOpen: false,
        isLoading: false,
        error: null
      });
      
      const handleCheckoutEvent = useCallback((event: CheckoutEvent) => {
        switch (event.event_type) {
          case "checkout.opened":
            setCheckoutState(prev => ({
              ...prev,
              isOpen: true,
              isLoading: false,
              error: null
            }));
            break;
            
          case "checkout.closed":
            setCheckoutState(prev => ({
              ...prev,
              isOpen: false,
              isLoading: false
            }));
            break;
            
          case "checkout.redirect":
            // Payment successful, user will be redirected
            setCheckoutState(prev => ({
              ...prev,
              isLoading: true
            }));
            break;
            
          case "checkout.error":
            setCheckoutState(prev => ({
              ...prev,
              isOpen: false,
              isLoading: false,
              error: event.data?.message || "Payment failed"
            }));
            break;
        }
      }, []);
      
      useEffect(() => {
        DodoPayments.Initialize({
          mode: "test",
          onEvent: handleCheckoutEvent,
          displayType: "overlay"
        });
      }, [handleCheckoutEvent]);
      
      return checkoutState;
    };
    ```

### Event Types Reference

<AccordionGroup>
  <Accordion title="checkout.opened">
    **Triggered when:** The checkout overlay is successfully opened and displayed to the user.
    
    **Event Data:** Usually minimal or empty
    
    **Use Cases:**
    - Track checkout initiation in analytics
    - Show loading states
    - Hide other UI elements
    - Start user interaction timers
  </Accordion>
  
  <Accordion title="checkout.closed">
    **Triggered when:** The user closes the checkout overlay without completing payment.
    
    **Event Data:** May include reason for closure
    
    **Use Cases:**
    - Track checkout abandonment
    - Reset UI states
    - Show abandoned cart reminders
    - Clean up any checkout-related data
  </Accordion>
  
  <Accordion title="checkout.redirect">
    **Triggered when:** Payment is successful and the user will be redirected to the success URL.
    
    **Event Data:**
    ```typescript
    {
      url: string; // The redirect URL
      payment_id?: string; // Payment identifier
    }
    ```
    
    **Use Cases:**
    - Track successful payments
    - Show success messages
    - Update cart state
    - Trigger conversion tracking
  </Accordion>
  
  <Accordion title="checkout.error">
    **Triggered when:** An error occurs during the checkout process.
    
    **Event Data:**
    ```typescript
    {
      message: string; // Error description
      code?: string; // Error code if available
      details?: any; // Additional error details
    }
    ```
    
    **Common Error Scenarios:**
    - Invalid product IDs
    - Network connectivity issues
    - Payment processing failures
    - Configuration errors
    
    **Use Cases:**
    - Display user-friendly error messages
    - Log errors for debugging
    - Track error rates
    - Implement retry mechanisms
  </Accordion>
</AccordionGroup>

### Best Practices for Event Handling

<Note>
- **Always implement error handling** for the `checkout.error` event
- **Track events in analytics** to understand user behavior
- **Provide user feedback** for all state changes
- **Handle network failures gracefully** with retry mechanisms
- **Test all event scenarios** during development
- **Keep event handlers lightweight** to avoid blocking the UI
</Note>

## Step-by-Step Guide

### 1. Project Setup

First, ensure you have a modern JavaScript/TypeScript project. We recommend using Next.js, React, or Vue.js.

```bash
# Using npm
npm install dodopayments-checkout

yarn add dodopayments-checkout
```

### 3. Basic Implementation

Create a new component for your checkout button:

```typescript
// components/CheckoutButton.tsx
"use client";

import { Button } from "@/components/ui/button";
import { DodoPayments } from "dodopayments-checkout";
import { useEffect, useState } from "react";

export function CheckoutButton() {
  const [isLoading, setIsLoading] = useState(false);

  useEffect(() => {
    // Initialize the SDK
    DodoPayments.Initialize({
      mode: "test", // Change to 'live' for production
      displayType: "overlay",
      theme: "dark",
      onEvent: (event) => {
        console.log("Checkout event:", event);
        
        // Handle different events
        switch (event.event_type) {
          case "checkout.opened":
            setIsLoading(false);
            break;
          case "checkout.error":
            // Handle error
            break;
        }
      },
    });
  }, []);

  const handleCheckout = async () => {
    try {
      setIsLoading(true);
      await DodoPayments.Checkout.open({
        products: [
          {
            productId: "pdt_your_product_id",
            quantity: 1,
          },
        ],
        redirectUrl: `${window.location.origin}/success`,
        queryParams: {
          email: "customer@example.com",
          disableEmail: "true",
        },
      });
    } catch (error) {
      console.error("Failed to open checkout:", error);
      setIsLoading(false);
    }
  };

  return (
    <Button 
      onClick={handleCheckout}
      disabled={isLoading}
    >
      {isLoading ? "Loading..." : "Checkout Now"}
    </Button>
  );
}
```
