## Python

The Python SDK offers a Pythonic interface to the Dodo Payments API with both synchronous and asynchronous support.

<Tabs>
  <Tab title="Installation">
    Install using pip or your preferred package manager:
    
    ```bash
    pip install dodopayments
    # or
    pip install dodopayments[async]  # for async support
    # or
    poetry add dodopayments
    ```
  </Tab>
  
  <Tab title="Synchronous Usage">
    Basic synchronous API usage:
    
    ```python
    from dodopayments import DodoPayments
    
    # Initialize client
    dodo = DodoPayments(
        api_key=os.getenv('DODO_SECRET_KEY'),
        environment='test'  # or 'live'
    )
    
    # Create a payment
    payment = dodo.payments.create(
        amount=2999,  # $29.99 in cents
        currency='USD',
        customer={
            'email': 'customer@example.com'
        },
        metadata={
            'order_id': 'order_123'
        }
    )
    
    print(f'Payment created: {payment.id}')
    
    # Retrieve payment
    retrieved_payment = dodo.payments.retrieve(payment.id)
    print(f'Payment status: {retrieved_payment.status}')
    ```
  </Tab>
  
  <Tab title="Asynchronous Usage">
    Async/await support for high-performance applications:
    
    ```python
    import asyncio
    from dodopayments import AsyncDodoPayments
    
    async def main():
        # Initialize async client
        dodo = AsyncDodoPayments(
            api_key=os.getenv('DODO_SECRET_KEY'),
            environment='test'
        )
        
        # Create multiple payments concurrently
        tasks = [
            dodo.payments.create(
                amount=1000 * (i + 1),
                currency='USD',
                customer={'email': f'customer{i}@example.com'}
            )
            for i in range(5)
        ]
        
        payments = await asyncio.gather(*tasks)
        print(f'Created {len(payments)} payments')
        
        # Close the client
        await dodo.close()
    
    # Run async code
    asyncio.run(main())
    ```
  </Tab>
  
  <Tab title="Error Handling">
    Comprehensive error handling patterns:
    
    ```python
    from dodopayments import DodoPayments, DodoError, InvalidRequestError
    
    dodo = DodoPayments(api_key=os.getenv('DODO_SECRET_KEY'))
    
    try:
        payment = dodo.payments.create(
            amount=2999,
            currency='USD',
            customer={'email': 'invalid-email'}
        )
    except InvalidRequestError as e:
        print(f'Invalid request: {e.message}')
        print(f'Error code: {e.code}')
        print(f'Parameter: {e.param}')
    except DodoError as e:
        print(f'Dodo API error: {e.message}')
    except Exception as e:
        print(f'Unexpected error: {e}')
    ```
  </Tab>
</Tabs>

## Go

The Go SDK provides a clean and idiomatic Go interface for integrating with Dodo Payments with full context support.

<Tabs>
  <Tab title="Installation">
    ```bash
    go get github.com/dodopayments/dodopayments-go
    ```
    
    **Requirements:**
    - Go 1.19 or higher
    - Standard library net/http package
  </Tab>
  
  <Tab title="Basic Usage">
    ```go
    package main
    
    import (
        "context"
        "fmt"
        "log"
        "os"
        
        "github.com/dodopayments/dodopayments-go"
    )
    
    func main() {
        // Initialize client
        client := dodopayments.New(dodopayments.Config{
            APIKey:      os.Getenv("DODO_SECRET_KEY"),
            Environment: dodopayments.EnvironmentTest, // or EnvironmentLive
        })
        
        ctx := context.Background()
        
        // Create a payment
        payment, err := client.Payments.Create(ctx, &dodopayments.PaymentCreateParams{
            Amount:   2999, // $29.99 in cents
            Currency: "USD",
            Customer: &dodopayments.CustomerParams{
                Email: "customer@example.com",
            },
            Metadata: map[string]string{
                "order_id": "order_123",
            },
        })
        
        if err != nil {
            log.Fatalf("Error creating payment: %v", err)
        }
        
        fmt.Printf("Payment created: %s\n", payment.ID)
    }
    ```
  </Tab>
  
  <Tab title="Advanced Features">

    import kotlinx.coroutines.*
    import kotlinx.coroutines.flow.*
    
    class PaymentService(private val dodo: DodoPayments) {
        
        // Suspend function for async payment creation
        suspend fun createPayment(request: PaymentRequest): Payment {
            return dodo.payments.create {
                amount = request.amount
                currency = request.currency
                customer {
                    email = request.customerEmail
                }
            }
        }
        
        // Flow for streaming payment updates
        fun getPaymentUpdates(paymentId: String): Flow<Payment> = flow {
            while (true) {
                val payment = dodo.payments.retrieve(paymentId)
                emit(payment)
                
                if (payment.status in listOf("succeeded", "failed", "cancelled")) {
                    break
                }
                
                delay(5000) // Poll every 5 seconds
            }
        }
        
        // Parallel payment processing
        suspend fun createMultiplePayments(requests: List<PaymentRequest>): List<Payment> {
            return requests.map { request ->
                async {
                    createPayment(request)
                }
            }.awaitAll()
        }
    }
    
    // Usage with structured concurrency
    suspend fun processPayments() = coroutineScope {
        val paymentService = PaymentService(dodo)
        
        // Create payments concurrently
        val payments = paymentService.createMultiplePayments(requests)
        
        // Monitor payment status
        payments.forEach { payment ->
            launch {
                paymentService.getPaymentUpdates(payment.id)
                    .collect { updatedPayment ->
                        println("Payment ${updatedPayment.id} status: ${updatedPayment.status}")
                    }
            }
        }
    }

    @Transactional
    class PaymentService(
        private val dodoPayments: DodoPayments,
        private val paymentRepository: PaymentRepository
    ) {
        
        suspend fun createPayment(request: PaymentRequest): Payment {
            return try {
                val dodoPayment = dodoPayments.payments.create {
                    amount = request.amount
                    currency = request.currency
                    customer {
                        email = request.customerEmail
                    }
                }
                
                // Save to local database
                val entity = PaymentEntity(
                    dodoPaymentId = dodoPayment.id,
                    amount = dodoPayment.amount,
                    status = dodoPayment.status
                )
                paymentRepository.save(entity)
                
                dodoPayment
                
            } catch (e: DodoException) {
                logger.error("Payment creation failed", e)
                throw PaymentProcessingException("Failed to create payment", e)
            }
        }
    }

## Best Practices

<AccordionGroup>
  <Accordion title="Security Guidelines">
    **API Key Management:**
    - Store API keys in environment variables, never in code
    - Use different keys for test and live environments
    - Rotate keys regularly and monitor for unauthorized usage
    - Implement proper access controls in production
    
    **Webhook Security:**
    - Always verify webhook signatures using SDK utilities
    - Use HTTPS endpoints for webhook URLs
    - Implement idempotency to handle duplicate events
    - Log webhook events for debugging and audit trails
  </Accordion>
  
  <Accordion title="Error Handling">
    **Robust Error Management:**
    - Implement retry logic with exponential backoff
    - Handle network timeouts gracefully
    - Log errors with sufficient context for debugging
    - Provide meaningful error messages to users
    
    **Common Error Scenarios:**
    - Rate limiting (429 status codes)
    - Authentication failures (401 status codes)
    - Invalid requests (400 status codes)
    - Server errors (5xx status codes)
  </Accordion>
  
  <Accordion title="Performance Optimization">

    - Implement caching for frequently accessed data
    - Monitor API response times and optimize accordingly
    
    **Resource Management:**
    - Reuse HTTP connections where possible
    - Set appropriate timeouts for API requests
    - Use async operations for high-throughput scenarios
    - Monitor memory usage in long-running applications
